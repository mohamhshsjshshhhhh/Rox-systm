التوثيق

📁 هيكل الملفات:

bot/
├── index.js          ← البوت الرئيسي
├── config.js         ← ملف الإعدادات
├── commands/         ← مجلد الأوامر
│   ├── moderation/
│   ├── general/
│   └── admin/
├── events/           ← مجلد الأحداث
├── handlers/         ← معالجات البوت
└── dashboard/        ← ملفات الداشبورد
    ├── index.html
    ├── style.css
    └── app.js

🔌 ربط الداشبورد مع البوت:

  1. أضف الكود التالي في index.js:
// في نهاية ملف index.js
const express = require('express');
const app = express();
const cors = require('cors');

app.use(cors());
app.use(express.json());

// API للداشبورد
app.get('/api/stats', (req, res) => {
    res.json({
        guilds: client.guilds.cache.size,
        users: client.users.cache.size,
        commands: client.commands.size,
        uptime: client.uptime
    });
});

app.post('/api/command', (req, res) => {
    // تنفيذ الأوامر من الداشبورد
    const { command, args } = req.body;
    // ... منطق التنفيذ
});

app.listen(3000, () => {
    console.log('Dashboard API running on port 3000');
});
